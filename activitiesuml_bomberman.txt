	:spawnCharacter;

	:findSomething;
	fork
		:buff;
		fork
			:morespeed;
		fork again
			:invulnerability;
		fork again
			:moreDetDamage;
		fork again
			:moreDetRange;
		end fork		
	fork again
		:obstacle;
		if (break?) then (yes)
			:putBomb;
			if (moveOutOfRange) then (True)
			else (False)
				:die;
			endif
		else (no)
		:moveTowardsDiferentDirection;
	endif
	fork again
		:enemy;
		fork 
			if (putBomb) then (yes)
			if (moveOutofRange) then (True)
				
			else (False)
				:die;
			endif
			else (no)
				:moveTowardsDiferentDirection;
			endif
		fork again
			if (getHit) then (yes)
				:die;
			else (no)
			
			endif		
		end fork
end fork
	:encountersNextLevelBlock;
	:PassLevel;
	endwhile (dead)
	stop
